图像转文字

Created at 2018-11-21 Updated at 2018-11-21 Tag 杂

def convert(img):
ascii_char=list(“$@B%8&WM#*oahkbdpqwmZO0QLCJUYXzcvunxrjft/|()1{}[]?-_+~ <>i!lI;:,\”^`’.”)

#字符长度
length=len(ascii_char)
img=Image.open(img)
(width,height)=img.size
#对图像进行缩小
img=img.resize((int(width*0.16),int(height*0.08)))
#转为灰度图像
img=img.convert('L')
txt=""
for i in range(img.size[1]):
    for j in range(img.size[0]):
        #获取每个坐标像素点的灰度
        gray=img.getpixel((j,i))
        #获取对应坐标的字符值
        unit=256.0 / length
        txt += ascii_char[int(gray / unit)]
    txt +='\n'
return  txt

#传入需要转化 的原始图片
txt=convert(‘4.png’)

#把转化后的字符存入txt文件
f=open(‘convert4.txt’,’w’)
f.write(txt)
f.close()
